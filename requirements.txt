# ROAR Academy — Python 3.11 environment
# This file is the canonical package list for the course.
# It mirrors the versions called out in README.md.
#
# Recommended setup (see README.md for the long-form instructions):
#     conda create -n roar python=3.11
#     conda activate roar
#     python -m pip install --upgrade pip
#     python -m pip install -r requirements.txt
#
# macOS Apple Silicon (M1/M2/M3/M4) users: the `tensorflow-macos` /
# `tensorflow-metal` lines below replace `tensorflow` automatically when pip
# evaluates the environment markers. See:
# https://developer.apple.com/metal/tensorflow-plugin/

# ---- core scientific stack ----
# TensorFlow 2.15 supports NumPy 1.23 through 1.26 (i.e. anything below 2.0).
numpy>=1.23,<2.0
matplotlib>=3.7,<3.10
scipy>=1.11
pandas>=2.0
pillow>=10.0
opencv-python>=4.8

# ---- deep learning (Keras 2 API, bundled inside TensorFlow 2.15) ----
# TF 2.15 is the last TensorFlow release that ships Keras 2 by default.
# Do NOT also `pip install keras` — TF 2.15 already installs Keras 2.15 as a
# dependency, and `import keras` and `from tensorflow import keras` both
# resolve to that copy. (TF 2.16+ would switch to Keras 3, which has subtle
# API differences this course is not yet ported to.)
#
# TF 2.15 publishes native macOS arm64 wheels under the regular `tensorflow`
# name, so a single line works on Linux, Windows, Intel Mac, and Apple Silicon.
# On Apple Silicon we additionally install Apple's Metal GPU plugin.
# (tensorflow-metal wheels are arm64-only; the marker keeps Intel Macs clean.)
tensorflow==2.15.*
tensorflow-metal==1.1.0 ; sys_platform == "darwin" and platform_machine == "arm64"

# ---- reinforcement learning ----
# Gymnasium is the maintained successor to OpenAI Gym. The samples import it via
# `import gymnasium as gym`, so all `gym.make('CartPole-v1')` calls keep working
# unchanged. Pygame is the rendering backend Gymnasium uses for CartPole.
gymnasium>=0.29,<2.0
pygame>=2.5

# Stable-Baselines3 and highway-env power the parking_sac_her.py sample (SAC + HER
# on the highway-env parking environment). SB3 is PyTorch-based, so `torch` is
# pulled in as a transitive dependency — on Apple Silicon and on Linux without
# CUDA, the default PyPI torch wheel installs a CPU build, no extra setup needed.
stable-baselines3>=2.3,<3.0
highway-env>=1.10,<2.0

# ---- notebook tooling for the .ipynb lessons in Part One / Part Two ----
jupyter>=1.0
ipykernel>=6.25
notebook>=7.0
ipywidgets>=8.0
